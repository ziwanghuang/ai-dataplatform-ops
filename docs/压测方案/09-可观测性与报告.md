# 09 — 可观测性验证与压测报告

> 本章定义如何在压测期间**验证可观测性系统本身**，以及如何生成结构化的压测报告。
> 可观测性不只是"能看到指标"，而是"指标值与实际行为一致"。

---

## 1. Prometheus 指标验证（P0）

### 1.1 为什么要验证指标

压测的价值不仅在于发现性能瓶颈，还要确认**监控体系能正确反映系统状态**。
如果压测时 P99 延迟已经 10s，但 Grafana 上只显示 2s，那监控就失去了意义。

### 1.2 指标完整性检查

```python
# tests/perf/verify_metrics.py
"""
压测期间 Prometheus 指标完整性验证.

验证所有 aiops_* 和 http_* 指标是否正确暴露和更新。
"""
import asyncio
import httpx

EXPECTED_METRICS = {
    # Agent 层 (15 个)
    "aiops_agent_invocations_total": {"type": "counter", "labels": ["agent", "status"]},
    "aiops_agent_duration_seconds": {"type": "histogram", "labels": ["agent"]},
    "aiops_agent_tokens_total": {"type": "counter", "labels": ["agent", "model"]},
    "aiops_rag_retrieval_duration_seconds": {"type": "histogram", "labels": ["retriever_type"]},
    "aiops_rag_results_count": {"type": "histogram", "labels": ["retriever_type"]},
    "aiops_rag_reranker_duration_seconds": {"type": "histogram"},
    "aiops_rag_cache_hit_total": {"type": "counter"},
    "aiops_rag_cache_miss_total": {"type": "counter"},
    "aiops_llm_call_duration_seconds": {"type": "histogram", "labels": ["model", "task_type"]},
    "aiops_llm_call_total": {"type": "counter", "labels": ["model", "status"]},
    "aiops_llm_token_usage_total": {"type": "counter", "labels": ["model", "token_type"]},
    "aiops_mcp_tool_call_duration_seconds": {"type": "histogram", "labels": ["tool_name", "status"]},
    "aiops_mcp_tool_call_total": {"type": "counter", "labels": ["tool_name", "status"]},
    "aiops_triage_route_total": {"type": "counter", "labels": ["method", "route"]},
    "aiops_triage_rule_engine_hit_total": {"type": "counter"},

    # HTTP 层 (3 个)
    "http_requests_total": {"type": "counter", "labels": ["method", "path", "status"]},
    "http_request_duration_seconds": {"type": "histogram", "labels": ["method", "path"]},
    "http_requests_in_progress": {"type": "gauge", "labels": ["method", "path"]},
}


async def verify_metrics(base_url: str = "http://localhost:8080"):
    """验证所有预期指标是否存在且有数据."""
    async with httpx.AsyncClient(base_url=base_url, timeout=10) as client:
        resp = await client.get("/metrics")
        if resp.status_code != 200:
            print(f"❌ /metrics 端点不可用 ({resp.status_code})")
            return False

        text = resp.text
        all_ok = True
        missing = []
        present = []

        for metric_name, info in EXPECTED_METRICS.items():
            # 检查指标是否存在（counter 会有 _total 后缀，histogram 会有 _bucket 后缀）
            found = False
            for suffix in ["", "_total", "_bucket", "_sum", "_count"]:
                if f"{metric_name}{suffix}" in text:
                    found = True
                    break

            if found:
                present.append(metric_name)
            else:
                missing.append(metric_name)
                all_ok = False

        # 打印结果
        print(f"\n{'='*60}")
        print(f"Prometheus 指标验证报告")
        print(f"{'='*60}")
        print(f"预期指标: {len(EXPECTED_METRICS)}")
        print(f"已找到:   {len(present)} ✅")
        print(f"缺失:     {len(missing)} ❌")

        if missing:
            print(f"\n缺失指标:")
            for m in missing:
                print(f"  - {m}")

        print(f"\n结论: {'✅ 全部通过' if all_ok else '⚠️ 有指标缺失'}")
        return all_ok


async def verify_metric_values(base_url: str = "http://localhost:8080"):
    """验证指标值是否合理（需在压测期间运行）."""
    async with httpx.AsyncClient(base_url=base_url, timeout=10) as client:
        resp = await client.get("/metrics")
        metrics = parse_metrics(resp.text)

        checks = []

        # 检查 1: HTTP 请求数 > 0（说明中间件在工作）
        http_total = sum(
            v for k, v in metrics.items()
            if k.startswith("http_requests_total")
        )
        checks.append(("HTTP 请求总数 > 0", http_total > 0, f"实际: {http_total}"))

        # 检查 2: Agent 调用数 > 0
        agent_total = sum(
            v for k, v in metrics.items()
            if k.startswith("aiops_agent_invocations_total")
        )
        checks.append(("Agent 调用数 > 0", agent_total > 0, f"实际: {agent_total}"))

        # 检查 3: Triage 路由分布合理
        triage_direct = metrics.get(
            'aiops_triage_route_total{method="rule_engine",route="direct_tool"}', 0
        )
        triage_diag = metrics.get(
            'aiops_triage_route_total{method="llm",route="diagnosis"}', 0
        )
        total_triage = triage_direct + triage_diag
        if total_triage > 0:
            direct_ratio = triage_direct / total_triage * 100
            checks.append((
                "DirectTool 路由占比 30-60%",
                30 <= direct_ratio <= 60,
                f"实际: {direct_ratio:.1f}%",
            ))

        # 检查 4: LLM 调用延迟直方图有数据
        llm_count = sum(
            v for k, v in metrics.items()
            if k.startswith("aiops_llm_call_duration_seconds_count")
        )
        checks.append(("LLM 延迟直方图有数据", llm_count > 0, f"采样数: {llm_count}"))

        # 检查 5: HTTP 进行中请求 >= 0（不应为负数）
        in_progress = sum(
            v for k, v in metrics.items()
            if k.startswith("http_requests_in_progress")
        )
        checks.append(("HTTP 进行中请求 >= 0", in_progress >= 0, f"实际: {in_progress}"))

        # 打印结果
        print(f"\n{'='*60}")
        print(f"指标值合理性检查")
        print(f"{'='*60}")
        for name, passed, detail in checks:
            status = "✅" if passed else "❌"
            print(f"  {status} {name} — {detail}")


def parse_metrics(text: str) -> dict:
    """解析 Prometheus 文本格式为 dict."""
    result = {}
    for line in text.split("\n"):
        if line.startswith("#") or not line.strip():
            continue
        parts = line.rsplit(" ", 1)
        if len(parts) == 2:
            try:
                result[parts[0]] = float(parts[1])
            except ValueError:
                pass
    return result


if __name__ == "__main__":
    asyncio.run(verify_metrics())
    asyncio.run(verify_metric_values())
```

### 1.3 指标一致性验证

指标值必须与 k6 客户端统计一致（允许 ±5% 误差，因为采集时间窗口不同）：

| 对比项 | k6 侧 | Prometheus 侧 | 判定 |
|--------|--------|---------------|------|
| 总请求数 | `http_reqs` | `http_requests_total` | 差异 <5% |
| 成功率 | `http_req_failed` | `http_requests_total{status="500"}` | 一致 |
| P95 延迟 | `http_req_duration{p(95)}` | `histogram_quantile(0.95, http_request_duration_seconds_bucket)` | 差异 <20% |
| 进行中请求 | `vus` (近似) | `http_requests_in_progress` | 趋势一致 |

## 3. 告警规则验证（P1）

### 3.1 压测应触发的告警

在高负载和故障注入场景下，验证告警规则是否正确触发：

| 告警规则 | 触发条件 | 压测场景 | 预期 |
|---------|---------|---------|------|
| `AgentHighLatency` | P95 >10s 持续 5m | 高负载 E2E | 触发 |
| `AgentHighErrorRate` | 错误率 >5% 持续 3m | LLM 故障注入 | 触发 |
| `LLMProviderDown` | 某 provider 错误率 100% 持续 2m | Provider 熔断测试 | 触发 |
| `HighMemoryUsage` | RSS >1GB 持续 10m | 长跑测试 | 不应触发 |
| `MCPServerDown` | MCP 连接失败 >10 次/分钟 | MCP 故障注入 | 触发 |

### 3.2 验证脚本

```python
# tests/perf/verify_alerts.py
"""
验证 Prometheus 告警规则在压测期间正确触发.

需要 Prometheus HTTP API 可用 (默认 http://localhost:9090)
"""
import asyncio
import httpx

PROMETHEUS_URL = "http://localhost:9090"


async def check_firing_alerts():
    """检查当前触发中的告警."""
    async with httpx.AsyncClient(timeout=10) as client:
        # 查询 ALERTS 指标
        resp = await client.get(
            f"{PROMETHEUS_URL}/api/v1/query",
            params={"query": "ALERTS{alertstate='firing'}"},
        )
        if resp.status_code != 200:
            print(f"❌ Prometheus API 不可用 ({resp.status_code})")
            return

        data = resp.json()
        results = data.get("data", {}).get("result", [])

        print(f"\n{'='*60}")
        print(f"当前触发中的告警")
        print(f"{'='*60}")

        if not results:
            print("  (无告警触发)")
        else:
            for alert in results:
                labels = alert.get("metric", {})
                name = labels.get("alertname", "unknown")
                severity = labels.get("severity", "unknown")
                print(f"  🔔 {name} [{severity}]")
                for k, v in labels.items():
                    if k not in ("alertname", "severity", "alertstate", "__name__"):
                        print(f"      {k}: {v}")


async def check_alert_history(alert_name: str, hours: int = 1):
    """检查过去 N 小时内某告警的触发历史."""
    async with httpx.AsyncClient(timeout=10) as client:
        resp = await client.get(
            f"{PROMETHEUS_URL}/api/v1/query_range",
            params={
                "query": f'ALERTS{{alertname="{alert_name}"}}',
                "start": f"-{hours}h",
                "end": "now",
                "step": "60s",
            },
        )
        if resp.status_code == 200:
            data = resp.json()
            results = data.get("data", {}).get("result", [])
            print(f"\n{alert_name} 过去 {hours}h 触发记录: {len(results)} 条")


if __name__ == "__main__":
    asyncio.run(check_firing_alerts())
```

### 4.2 自动报告生成脚本

```python
# tests/perf/generate_report.py
"""
从 k6 JSON 结果自动生成压测报告.

用法:
  python tests/perf/generate_report.py results/

会读取 results/ 下所有 *.json 文件，汇总为 Markdown 报告。
"""
import json
import sys
from datetime import datetime
from pathlib import Path


def load_k6_results(results_dir: str) -> dict:
    """加载所有 k6 JSON 结果."""
    results = {}
    for f in Path(results_dir).glob("*.json"):
        with open(f) as fp:
            try:
                data = json.load(fp)
                results[f.stem] = data
            except json.JSONDecodeError:
                print(f"跳过无效文件: {f}")
    return results


def extract_metric(data: dict, metric_name: str, stat: str = "p(95)") -> float:
    """从 k6 JSON 提取指标值."""
    metrics = data.get("metrics", {})
    metric = metrics.get(metric_name, {})
    values = metric.get("values", {})
    return values.get(stat, 0)


def generate_report(results: dict) -> str:
    """生成 Markdown 报告."""
    lines = [
        f"# AI DataPlatform Ops Agent — 压测报告",
        f"",
        f"> 自动生成于 {datetime.now().strftime('%Y-%m-%d %H:%M')}",
        f"",
        f"## 执行摘要",
        f"",
    ]

    # 汇总各场景
    for name, data in sorted(results.items()):
        lines.append(f"### {name}")
        lines.append("")

        # 提取通用指标
        metrics = data.get("metrics", {})
        for metric_name, metric_data in sorted(metrics.items()):
            values = metric_data.get("values", {})
            if values:
                line_parts = [f"  - **{metric_name}**:"]
                for stat, val in values.items():
                    if isinstance(val, float):
                        line_parts.append(f"{stat}={val:.2f}")
                    else:
                        line_parts.append(f"{stat}={val}")
                lines.append(" ".join(line_parts))
        lines.append("")

    # SLO 对照表
    lines.extend([
        "## SLO 达标情况",
        "",
        "| SLO | 目标 | 实际 | 状态 |",
        "|-----|------|------|------|",
    ])

    # 自动填充（如果有对应数据）
    slo_checks = [
        ("快速路径 P95", "<500ms", "e2e_fast_path", "fast_path_latency", "p(95)", 500),
        ("完整诊断 P95", "<15s", "e2e_full_diagnosis", "e2e_diagnosis_latency", "p(95)", 15000),
        ("告警接收 P95", "<2s", "e2e_alert_storm", "alert_receive_latency", "p(95)", 2000),
    ]

    for slo_name, target, result_key, metric, stat, threshold in slo_checks:
        if result_key in results:
            val = extract_metric(results[result_key], metric, stat)
            status = "✅" if val < threshold else "❌"
            lines.append(f"| {slo_name} | {target} | {val:.0f}ms | {status} |")
        else:
            lines.append(f"| {slo_name} | {target} | N/A | ⏭ |")

    lines.append("")
    return "\n".join(lines)


if __name__ == "__main__":
    if len(sys.argv) < 2:
        print("用法: python generate_report.py <results_dir>")
        sys.exit(1)

    results = load_k6_results(sys.argv[1])
    if not results:
        print("未找到 k6 结果文件")
        sys.exit(1)

    report = generate_report(results)
    output_file = Path(sys.argv[1]) / "压测报告.md"
    output_file.write_text(report, encoding="utf-8")
    print(f"报告已生成: {output_file}")
```

## 5. 性能回归检测（P1）

### 5.1 目标

在 CI/CD 中自动检测性能回归——每次代码变更后与基线对比。

### 5.2 基线管理

```yaml
# tests/perf/baseline.yaml
# 性能基线 — 每次发布后更新
version: "2026-03-31"
commit: "abc1234"

baselines:
  fast_path_p95_ms: 300
  full_diagnosis_p95_ms: 12000
  alert_receive_p95_ms: 1500
  health_check_p99_ms: 3
  memory_peak_mb: 256
  
  # 允许的劣化百分比
  regression_threshold_pct: 10
```

### 5.3 回归检测脚本

```python
# tests/perf/check_regression.py
"""
性能回归检测 — CI/CD 集成.

对比当前压测结果与基线，超过阈值则报错。
"""
import json
import sys
from pathlib import Path

import yaml


def check_regression(baseline_file: str, results_file: str) -> bool:
    """对比基线和当前结果，返回是否通过."""
    with open(baseline_file) as f:
        baseline = yaml.safe_load(f)

    with open(results_file) as f:
        results = json.load(f)

    threshold_pct = baseline.get("baselines", {}).get("regression_threshold_pct", 10)
    baselines = baseline.get("baselines", {})
    all_ok = True

    print(f"{'='*60}")
    print(f"性能回归检测")
    print(f"基线版本: {baseline.get('version')} ({baseline.get('commit')})")
    print(f"回归阈值: {threshold_pct}%")
    print(f"{'='*60}\n")

    checks = [
        ("快速路径 P95", "fast_path_p95_ms", "fast_path_latency", "p(95)"),
        ("完整诊断 P95", "full_diagnosis_p95_ms", "e2e_diagnosis_latency", "p(95)"),
        ("告警接收 P95", "alert_receive_p95_ms", "alert_receive_latency", "p(95)"),
    ]

    for name, baseline_key, metric_key, stat in checks:
        baseline_val = baselines.get(baseline_key, 0)
        metrics = results.get("metrics", {})
        current_val = metrics.get(metric_key, {}).get("values", {}).get(stat, 0)

        if baseline_val == 0:
            print(f"  ⏭ {name}: 无基线数据")
            continue

        change_pct = ((current_val - baseline_val) / baseline_val) * 100

        if change_pct > threshold_pct:
            print(f"  ❌ {name}: {baseline_val:.0f}ms → {current_val:.0f}ms (+{change_pct:.1f}%) — 回归!")
            all_ok = False
        elif change_pct < -threshold_pct:
            print(f"  🎉 {name}: {baseline_val:.0f}ms → {current_val:.0f}ms ({change_pct:.1f}%) — 优化!")
        else:
            print(f"  ✅ {name}: {baseline_val:.0f}ms → {current_val:.0f}ms ({change_pct:+.1f}%)")

    print(f"\n结论: {'✅ 无回归' if all_ok else '❌ 检测到性能回归!'}")
    return all_ok


if __name__ == "__main__":
    if len(sys.argv) < 3:
        print("用法: python check_regression.py <baseline.yaml> <results.json>")
        sys.exit(1)

    ok = check_regression(sys.argv[1], sys.argv[2])
    sys.exit(0 if ok else 1)
```

## 6. 高基数指标防护验证（P2）

### 6.1 问题背景

如果 path 标签没有做规范化，`/api/v1/agent/sessions/{uuid}` 会产生无限多的 label 组合，
导致 Prometheus TSDB 膨胀、查询变慢。

### 6.2 验证方法

```python
# tests/perf/verify_cardinality.py
"""
验证 HTTP 指标路径规范化 — 确认没有高基数标签.

预期:
- /api/v1/agent/sessions/{任意uuid} → /api/v1/agent/sessions/{id}
- /api/v1/approval/{任意id} → /api/v1/approval/{id}
"""
import asyncio
import httpx
import uuid


async def test_path_normalization():
    """发送带 UUID 的请求，检查指标标签是否规范化."""
    async with httpx.AsyncClient(base_url="http://localhost:8080", timeout=10) as client:
        # 发送 10 个不同 session_id 的请求
        for _ in range(10):
            sid = str(uuid.uuid4())
            await client.get(f"/api/v1/agent/sessions/{sid}")

        # 检查 /metrics 中的标签
        resp = await client.get("/metrics")
        text = resp.text

        # 计算 http_requests_total 中包含 sessions 的行数
        session_lines = [
            line for line in text.split("\n")
            if "http_requests_total" in line and "sessions" in line
        ]

        print(f"\n包含 sessions 的指标行数: {len(session_lines)}")

        # 如果规范化正确，应该只有 1-2 行（不同 method/status 组合）
        # 如果没有规范化，会有 10 行（每个 UUID 一行）
        if len(session_lines) <= 5:
            print("✅ 路径规范化正常 — 无高基数风险")
            for line in session_lines:
                print(f"  {line[:100]}")
        else:
            print("❌ 路径规范化失败 — 存在高基数风险!")
            print(f"  发现 {len(session_lines)} 个不同的路径标签")


if __name__ == "__main__":
    asyncio.run(test_path_normalization())
```

### 7.2 快速复盘 Checklist

每次压测后必须回答以下问题：

```
□ 所有 SLO 是否达标？
□ 有没有发现新的性能瓶颈？
□ 监控指标是否准确反映了实际情况？
□ 告警规则是否正确触发（或正确不触发）？
□ 内存/连接/Goroutine 有无泄漏？
□ 与上次压测相比是否有回归？
□ 结果是否已归档并更新基线？
```
