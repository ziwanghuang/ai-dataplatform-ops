# 08 — 端到端场景压测

> 本章从**用户视角**模拟完整业务场景，验证系统在真实工作流下的性能、正确性和稳定性。
> 与前几章单模块 benchmark 不同，这里每个测试都跨越多个子系统。

## 1. 完整诊断链路 E2E（P0）

### 1.1 目标

验证 `Triage → Planning → Diagnostic（自环 ≤5 轮）→ Report → KnowledgeSink` 完整链路的端到端延迟和正确性。

### 1.2 SLO 基准

| 指标 | 目标值 | 说明 |
|------|--------|------|
| P50 延迟 | < 8s | Mock LLM（100ms/call），3 轮诊断 |
| P95 延迟 | < 15s | 含自环 + RAG + MCP 调用 |
| P99 延迟 | < 30s | 极端情况（5 轮自环 + 重试） |
| 成功率 | > 99% | 必须返回有效诊断结果 |
| 报告完整性 | 100% | 每个成功请求都生成报告 |

## 3. 告警风暴场景（P0）

### 3.1 背景

生产环境的告警风暴：短时间内大量告警同时到达。如 HDFS DataNode 批量掉线可能在 1 分钟内触发 50+ 条告警。

### 3.2 SLO 基准

| 指标 | 目标值 | 说明 |
|------|--------|------|
| 接收率 | 100% | 所有告警都必须被接收（不丢告警） |
| 接收延迟 P95 | < 2s | 告警接收是异步的，应该很快返回 |
| 降级率 | < 10% | 最多 10% 走 accepted_degraded |
| 无 500 错误 | 0 | 告警端点永不返回 500 |

### 3.3 k6 脚本 — 告警风暴

### 4.3 Session 泄漏检测

```python
# tests/perf/check_session_leak.py
"""
压测后检查 session store 内存泄漏。

使用方式: 在多用户压测运行时，定期调用 /healthz 和 /metrics 检查内存。
"""
import asyncio
import httpx
import psutil
import time


async def monitor_memory(base_url: str, duration_seconds: int = 300):
    """监控 Agent 进程内存使用趋势."""
    samples: list[dict] = []
    start = time.monotonic()

    async with httpx.AsyncClient(base_url=base_url, timeout=10) as client:
        while time.monotonic() - start < duration_seconds:
            # 获取 /metrics 中的 process_resident_memory_bytes
            resp = await client.get("/metrics")
            if resp.status_code == 200:
                for line in resp.text.split("\n"):
                    if line.startswith("process_resident_memory_bytes"):
                        mem_bytes = float(line.split()[-1])
                        samples.append({
                            "time": time.monotonic() - start,
                            "rss_mb": mem_bytes / 1024 / 1024,
                        })
                        break

            await asyncio.sleep(5)  # 每 5 秒采样

    # 分析趋势
    if len(samples) > 10:
        first_10 = sum(s["rss_mb"] for s in samples[:10]) / 10
        last_10 = sum(s["rss_mb"] for s in samples[-10:]) / 10
        growth_mb = last_10 - first_10
        growth_pct = (growth_mb / first_10) * 100

        print(f"初始 RSS: {first_10:.1f} MB")
        print(f"最终 RSS: {last_10:.1f} MB")
        print(f"增长: {growth_mb:.1f} MB ({growth_pct:.1f}%)")

        if growth_pct > 50:
            print("⚠️ 警告: 内存增长 >50%，可能存在泄漏!")
        else:
            print("✅ 内存增长在合理范围内")


if __name__ == "__main__":
    asyncio.run(monitor_memory("http://localhost:8080"))
```

```python
1. 发送高风险诊断请求 → 触发 HITL Gate

        print(f"    Session: {session_id}")
        print(f"    Route: {data.get('route')}")

        # Step 2: 查询审批列表
        print("[2] 查询待审批列表...")
        approval_resp = await client.get("/api/v1/approval/pending")
        if approval_resp.status_code == 200:

            if final_resp.status_code == 200:
                final_data = final_resp.json()
                print(f"    最终状态: {final_data.get('status')}")

async def test_hitl_concurrent(concurrency: int = 10):
    """并发 HITL 流程压测."""
```

### 7.3 监控脚本 — 资源趋势

```python
# tests/perf/monitor_resources.py
"""
长跑测试期间的资源监控脚本.

采集: RSS 内存、CPU 使用率、打开文件数、连接数
输出: CSV 文件 + 趋势分析
"""
import asyncio
import csv
import time
from datetime import datetime
from pathlib import Path

import httpx


async def collect_metrics(base_url: str, output_file: str, interval: int = 10):
    """定期采集资源指标并写入 CSV."""
    output = Path(output_file)
    headers_written = output.exists()

    with open(output, "a", newline="") as f:
        writer = csv.writer(f)
        if not headers_written:
            writer.writerow([
                "timestamp", "rss_mb", "open_fds", "goroutines",
                "http_in_progress", "latency_p95_ms",
            ])

        async with httpx.AsyncClient(base_url=base_url, timeout=10) as client:
            while True:
                try:
                    resp = await client.get("/metrics")
                    if resp.status_code != 200:
                        await asyncio.sleep(interval)
                        continue

                    metrics = parse_prometheus_text(resp.text)
                    writer.writerow([
                        datetime.now().isoformat(),
                        metrics.get("process_resident_memory_bytes", 0) / 1024 / 1024,
                        metrics.get("process_open_fds", 0),
                        metrics.get("go_goroutines", 0),
                        metrics.get("http_requests_in_progress", 0),
                        metrics.get("http_request_duration_seconds_p95", 0) * 1000,
                    ])
                    f.flush()

                except Exception as e:
                    print(f"采集失败: {e}")

                await asyncio.sleep(interval)


def parse_prometheus_text(text: str) -> dict:
    """解析 Prometheus 文本格式."""
    result = {}
    for line in text.split("\n"):
        if line.startswith("#") or not line.strip():
            continue
        parts = line.split()
        if len(parts) >= 2:
            name = parts[0].split("{")[0]
            try:
                result[name] = float(parts[-1])
            except ValueError:
                pass
    return result


if __name__ == "__main__":
    asyncio.run(collect_metrics(
        "http://localhost:8080",
        "results/endurance_metrics.csv",
        interval=10,
    ))
```
