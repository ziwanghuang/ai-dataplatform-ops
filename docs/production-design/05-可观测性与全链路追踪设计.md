# 05 - 可观测性与全链路追踪设计

> **本文档定义**：如何让 AIOps 系统本身变得"可观测"——追踪每次 Agent 执行的完整链路、评测系统效果、管理 Bad Case。
> **核心理念**：对 AI 系统来说，可观测性不是"锦上添花"，而是"生存必需品"——不可观测的 AI 系统等于定时炸弹。

## 二、全链路追踪（Tracing）

### 2.1 OpenTelemetry 集成

```python
from opentelemetry import trace
from opentelemetry.sdk.trace import TracerProvider
from opentelemetry.exporter.otlp.proto.grpc.trace_exporter import OTLPSpanExporter

# 初始化 Tracer
provider = TracerProvider(resource=Resource.create({
    "service.name": "aiops-agent",
    "service.version": "1.0.0",
    "deployment.environment": "production",
}))
provider.add_span_processor(BatchSpanProcessor(OTLPSpanExporter()))
trace.set_tracer_provider(provider)
tracer = trace.get_tracer("aiops.agent")


# Agent 执行的 Trace 示例
async def run_agent_with_tracing(state: AgentState):
    with tracer.start_as_current_span("agent.session",
        attributes={
            "agent.request_id": state["request_id"],
            "agent.request_type": state["request_type"],
            "agent.cluster": state["cluster_id"],
            "agent.user": state["user_id"],
        }
    ) as session_span:

        # Triage
        with tracer.start_as_current_span("agent.triage") as triage_span:
            state = await triage_agent(state)
            triage_span.set_attribute("triage.intent", state["intent"])
            triage_span.set_attribute("triage.complexity", state["complexity"])
            triage_span.set_attribute("triage.route", state["route"])

        # Planning
        with tracer.start_as_current_span("agent.planning") as plan_span:
            state = await planning_agent(state)
            plan_span.set_attribute("planning.steps_count", len(state["task_plan"]))
            plan_span.set_attribute("planning.rag_hits", len(state.get("rag_context", [])))

        # Diagnostic (可能多轮)
        for round_num in range(state["max_collection_rounds"]):
            with tracer.start_as_current_span(f"agent.diagnostic.round_{round_num}") as diag_span:
                # 工具调用
                for tool_call in state["task_plan"]:
                    with tracer.start_as_current_span(f"tool.{tool_call['tool']}") as tool_span:
                        result = await mcp_client.call_tool(tool_call["tool"], tool_call["params"])
                        tool_span.set_attribute("tool.name", tool_call["tool"])
                        tool_span.set_attribute("tool.result_size", len(str(result)))

                # LLM 分析
                with tracer.start_as_current_span("llm.diagnostic_analysis") as llm_span:
                    diagnosis = await diagnostic_llm(state)
                    llm_span.set_attribute("llm.model", diagnosis.model)
                    llm_span.set_attribute("llm.tokens.prompt", diagnosis.usage.prompt_tokens)
                    llm_span.set_attribute("llm.tokens.completion", diagnosis.usage.completion_tokens)
                    llm_span.set_attribute("diagnosis.confidence", diagnosis.confidence)
                    llm_span.set_attribute("diagnosis.severity", diagnosis.severity)

        # Report
        with tracer.start_as_current_span("agent.report"):
            state = await report_agent(state)

        # 设置 Session 级别的聚合属性
        session_span.set_attribute("agent.total_tokens", state["total_tokens"])
        session_span.set_attribute("agent.total_cost_usd", state["total_cost_usd"])
        session_span.set_attribute("agent.tool_calls_count", len(state["tool_calls"]))
        session_span.set_attribute("agent.duration_ms",
            (datetime.now() - state["start_time"]).total_seconds() * 1000)
```

### 2.2 LLM 专用追踪（LangFuse 集成）

```python
from langfuse import Langfuse
from langfuse.decorators import observe, langfuse_context

langfuse = Langfuse(
    public_key="pk-...",
    secret_key="sk-...",
    host="https://langfuse.internal"  # 自部署
)

@observe(name="diagnostic_agent")
async def diagnostic_agent(state: AgentState) -> AgentState:
    # LangFuse 自动记录：
    # - 输入/输出
    # - 耗时
    # - Token 消耗
    # - 成本

    langfuse_context.update_current_observation(
        metadata={
            "request_id": state["request_id"],
            "cluster": state["cluster_id"],
            "component": state.get("intent", "unknown"),
        }
    )

    # LLM 调用会被自动追踪
    response = await llm_client.call(
        messages=[...],
        model="gpt-4o",
    )

    # 记录评分（用于后续评测）
    langfuse_context.score_current_trace(
        name="confidence",
        value=state["diagnosis"]["confidence"]
    )

    return state
```

### 3.2 告警规则

```yaml
# Prometheus AlertManager 规则

groups:
  - name: aiops_agent_alerts
    rules:
      # LLM 调用失败率过高
      - alert: HighLLMErrorRate
        expr: |
          rate(aiops_llm_calls_total{status="error"}[5m])
          / rate(aiops_llm_calls_total[5m]) > 0.1
        for: 5m
        labels:
          severity: critical
        annotations:
          summary: "LLM 调用失败率 > 10%"
          description: "Provider {{ $labels.provider }} 的失败率为 {{ $value | humanizePercentage }}"

      # Agent 请求延迟过高
      - alert: HighAgentLatency
        expr: |
          histogram_quantile(0.99, rate(aiops_agent_request_duration_seconds_bucket[5m])) > 30
        for: 5m
        labels:
          severity: warning
        annotations:
          summary: "Agent P99 延迟 > 30s"

      # Token 成本超预算
      - alert: HighTokenCost
        expr: |
          increase(aiops_llm_cost_usd_total[1h]) > 50
        for: 1h
        labels:
          severity: warning
        annotations:
          summary: "过去 1 小时 LLM 成本 > $50"

      # 诊断置信度过低
      - alert: LowDiagnosisConfidence
        expr: |
          histogram_quantile(0.5, rate(aiops_diagnosis_confidence_bucket[1h])) < 0.5
        for: 1h
        labels:
          severity: warning
        annotations:
          summary: "诊断中位置信度 < 0.5，可能存在模型退化"

      # 熔断器开启
      - alert: CircuitBreakerOpen
        expr: aiops_circuit_breaker_state > 0
        for: 1m
        labels:
          severity: critical
        annotations:
          summary: "工具 {{ $labels.tool_name }} 熔断器已开启"
```

### 4.2 评测指标定义

| 指标 | 计算方式 | 目标值 | 权重 |
|------|---------|--------|------|
| **Triage 准确率** | 意图识别正确数 / 总数 | >90% | 15% |
| **工具选择准确率** | 选对工具的诊断数 / 总诊断数 | >85% | 20% |
| **诊断准确率** | 根因正确的诊断数 / 总诊断数 | >75% | 25% |
| **RAG Faithfulness** | 回答被检索文档支撑的比例 | >85% | 15% |
| **RAG Context Precision** | 检索到的文档中真正相关的比例 | >70% | 10% |
| **格式合规率** | 输出通过 Schema 验证的比例 | >98% | 5% |
| **安全测试通过率** | 安全用例全部拒绝 | 100% | 10% |

class BadCaseTracker:
    """Bad Case 自动追踪和分析"""

    CATEGORIES = {
        "wrong_diagnosis":    "诊断结论错误",
        "missed_root_cause":  "遗漏根因",
        "wrong_tool":         "工具选择错误",
        "hallucination":      "LLM 幻觉",
        "format_error":       "输出格式错误",
        "rag_miss":           "RAG 未检索到关键文档",
        "unsafe_suggestion":  "不安全的修复建议",
        "timeout":            "执行超时",
        "user_confusion":     "用户无法理解输出",
    }

    async def auto_detect(self, state: AgentState, user_feedback: dict = None):
        """自动检测可能的 Bad Case"""
        issues = []

        # 1. 低置信度
        if state["diagnosis"]["confidence"] < 0.5:
            issues.append(("low_confidence", f"置信度仅 {state['diagnosis']['confidence']:.0%}"))

        # 2. HITL 被拒绝
        if state.get("hitl_status") == "rejected":
            issues.append(("rejected_suggestion", f"修复建议被拒绝: {state.get('hitl_comment')}"))

        # 3. 用户负反馈
        if user_feedback and user_feedback.get("rating") == "negative":
            issues.append(("user_negative", user_feedback.get("comment", "用户给出负面反馈")))

        # 4. 格式校验失败
        if state.get("error_count", 0) > 0:
            issues.append(("format_error", f"执行过程中发生 {state['error_count']} 次错误"))

        if issues:
            await self._create_bad_case(state, issues)

### 6.1 Token 成本追踪

```python
class CostTracker:
    """
    精细化 Token 成本追踪

    每次 LLM 调用自动计算成本并累加
    """

    PRICING = {
        # model_name: (input_price_per_1m_tokens, output_price_per_1m_tokens)
        "gpt-4o":           (2.50, 10.00),
        "gpt-4o-mini":      (0.15, 0.60),
        "claude-3.5-sonnet": (3.00, 15.00),
        "deepseek-v3":      (0.27, 1.10),
        "qwen2.5-72b-local": (0.00, 0.00),  # 自部署，按 GPU 时间算
    }

    def calculate_cost(self, model: str, prompt_tokens: int, completion_tokens: int) -> float:
        prices = self.PRICING.get(model, (5.0, 15.0))  # 默认用高价估算
        input_cost = prompt_tokens / 1_000_000 * prices[0]
        output_cost = completion_tokens / 1_000_000 * prices[1]
        return input_cost + output_cost

    async def track(self, request_id: str, model: str, usage: TokenUsage):
        cost = self.calculate_cost(model, usage.prompt_tokens, usage.completion_tokens)

        # 更新 Prometheus 指标
        metrics.llm_cost_usd_total.labels(provider=..., model=model).inc(cost)
        metrics.llm_tokens_total.labels(model=model, token_type="prompt").inc(usage.prompt_tokens)
        metrics.llm_tokens_total.labels(model=model, token_type="completion").inc(usage.completion_tokens)

        # 记录到数据库（用于细粒度分析）
        await self.db.execute("""
            INSERT INTO llm_cost_log (request_id, model, prompt_tokens, completion_tokens, cost_usd, timestamp)
            VALUES ($1, $2, $3, $4, $5, NOW())
        """, request_id, model, usage.prompt_tokens, usage.completion_tokens, cost)
```
