#### 改进 A：LLM 不确定性治理 — Self-Consistency 缺少具体实现

**现状**：06-安全防护 §6.1 提到 Self-Consistency（多次采样取一致性最高结果），但仅作为概念提及，无实现方案。

**详细优化方案**：

```python
class SelfConsistencyDiagnosis:
    """
    关键诊断场景的 Self-Consistency 策略
    
    仅对高影响场景启用（成本考虑）：
    - severity = critical 的告警
    - 置信度在 0.5-0.8 的边界区间
    """
    
    SAMPLE_COUNT = 3  # 采样次数（成本与精度平衡）
    CONSISTENCY_THRESHOLD = 2  # 至少 2/3 一致才采用
    
    async def diagnose_with_consistency(
        self, state: AgentState, llm_client: LLMClient
    ) -> DiagnosticOutput:
        # 1. 多次采样（temperature=0.3 引入微小差异）
        samples = await asyncio.gather(*[
            self._single_diagnosis(state, llm_client, temperature=0.3)
            for _ in range(self.SAMPLE_COUNT)
        ])
        
        # 2. 聚类根因描述（基于语义相似度）
        root_cause_groups = self._cluster_root_causes(
            [s.root_cause for s in samples], similarity_threshold=0.85
        )
        
        # 3. 选择最大聚类的结果
        largest_cluster = max(root_cause_groups, key=lambda g: len(g.members))
        
        if len(largest_cluster.members) >= self.CONSISTENCY_THRESHOLD:
            # 多数一致 → 取该聚类中置信度最高的样本
            best = max(largest_cluster.members, key=lambda s: s.confidence)
            best.confidence = min(best.confidence + 0.1, 1.0)  # 一致性加分
            return best
        else:
            # 无一致性 → 降低置信度，标记需人工确认
            best = max(samples, key=lambda s: s.confidence)
            best.confidence = max(best.confidence - 0.2, 0.1)
            best.root_cause = f"⚠️ [多次诊断不一致] {best.root_cause}"
            return best
    
    def _cluster_root_causes(self, causes: list[str], similarity_threshold: float):
        """基于 Embedding 余弦相似度聚类"""
        embeddings = self.embedding_model.encode(causes)
        clusters = []
        for i, emb in enumerate(embeddings):
            placed = False
            for cluster in clusters:
                if cosine_similarity(emb, cluster.centroid) > similarity_threshold:
                    cluster.add(causes[i], emb)
                    placed = True
                    break
            if not placed:
                clusters.append(Cluster(causes[i], emb))
        return clusters
```

**触发条件**：仅在以下场景启用，避免成本膨胀：
- 告警严重度 = critical
- 首次诊断置信度在 0.5-0.8 之间
- 预估额外成本 < $0.10/次

```python
class PromptInjectionClassifier:
    """
    基于微调 Transformer 的 Prompt 注入检测器
    
    架构：规则引擎（快速，0.01ms）→ ML 分类器（精确，5ms）→ LLM 审查（最终，200ms）
    """
    
    def __init__(self):
        # 使用 DeBERTa-v3-small 微调，二分类（注入/正常）
        # 训练数据：2000+ 注入样本 + 5000+ 正常运维查询
        self.model = AutoModelForSequenceClassification.from_pretrained(
            "models/prompt-injection-detector-v1"
        )
        self.tokenizer = AutoTokenizer.from_pretrained(
            "models/prompt-injection-detector-v1"
        )
        self.threshold = 0.85  # 高精度阈值，减少误报
    
    async def detect(self, user_input: str) -> InjectionResult:
        # Layer 1: 规则引擎（微秒级，捕获明显攻击）
        rule_result = self.rule_engine.check(user_input)
        if not rule_result.safe:
            return InjectionResult(is_injection=True, confidence=0.99, 
                                   method="rule", detail=rule_result.reason)
        
        # Layer 2: ML 分类器（毫秒级，捕获变体攻击）
        inputs = self.tokenizer(user_input, return_tensors="pt", 
                               truncation=True, max_length=512)
        with torch.no_grad():
            logits = self.model(**inputs).logits
            prob = torch.softmax(logits, dim=-1)[0][1].item()  # 注入概率
        
        if prob > self.threshold:
            return InjectionResult(is_injection=True, confidence=prob,
                                   method="ml_classifier", 
                                   detail=f"ML 检测注入概率: {prob:.2%}")
        
        # Layer 3: 灰度区间（0.5-0.85）→ 记录但放行，供后续分析
        if prob > 0.5:
            await self.log_suspicious(user_input, prob)
        
        return InjectionResult(is_injection=False, confidence=1-prob, method="safe")
    
    # 训练数据来源
    TRAINING_SOURCES = """
    正样本（注入攻击）：
    - OWASP LLM Top 10 攻击样本库
    - 运维场景定制注入样本（通过日志/知识库的间接注入）
    - 社会工程攻击样本（"我是管理员"类型）
    - 红队渗透测试收集的真实攻击
    
    负样本（正常查询）：
    - 真实运维人员的历史查询日志
    - 各组件的正常诊断请求
    - 包含技术术语的合法查询
    """
```

#### 改进 C：知识图谱 — 实体提取自动化

**现状**：04-RAG知识库 §3.3 中 `_extract_entities` 方法未给出具体实现，这是 Graph-RAG 的关键瓶颈。

**详细优化方案**：

```python
class OpsEntityExtractor:
    """
    运维领域实体提取器
    
    三层提取策略：规则 → NER 模型 → LLM（按需）
    """
    
    # Layer 1: 规则提取（零成本，覆盖 60% 场景）
    COMPONENT_PATTERNS = {
        r'\b(?:HDFS|NameNode|DataNode|NN|DN)\b': 'HDFS',
        r'\b(?:YARN|ResourceManager|NodeManager|RM|NM)\b': 'YARN',
        r'\b(?:Kafka|Broker|Consumer|Producer)\b': 'Kafka',
        r'\b(?:Elasticsearch|ES)\b': 'Elasticsearch',
        r'\b(?:ZooKeeper|ZK)\b': 'ZooKeeper',
        r'\b(?:HBase|RegionServer|RS)\b': 'HBase',
        r'\b(?:Impala|impalad|Catalog)\b': 'Impala',
    }
    
    FAULT_PATTERNS = {
        r'堆?内存[不高使用超过]': 'memory_issue',
        r'(?:OOM|OutOfMemory)': 'oom',
        r'(?:超时|timeout|Timeout)': 'timeout',
        r'(?:GC|垃圾回收|Full GC)': 'gc_issue',
        r'(?:磁盘[满使用]|disk.?(?:full|usage))': 'disk_issue',
        r'(?:副本不足|UnderReplicated)': 'under_replicated',
        r'(?:SafeMode|安全模式)': 'safe_mode',
        r'(?:消费延迟|consumer.?lag)': 'consumer_lag',
    }
    
    METRIC_PATTERNS = {
        r'(\d+(?:\.\d+)?)\s*%': 'percentage',
        r'(\d+(?:\.\d+)?)\s*(?:ms|毫秒)': 'latency_ms',
        r'(\d+(?:\.\d+)?)\s*(?:GB|MB|TB)': 'storage_size',
        r'(\d+(?:\.\d+)?)\s*(?:个|条|块)': 'count',
    }
    
    def extract(self, text: str) -> ExtractedEntities:
        entities = ExtractedEntities()
        
        # 规则提取组件
        for pattern, component in self.COMPONENT_PATTERNS.items():
            if re.search(pattern, text, re.IGNORECASE):
                entities.components.append(component)
        
        # 规则提取故障模式
        for pattern, fault in self.FAULT_PATTERNS.items():
            if re.search(pattern, text, re.IGNORECASE):
                entities.fault_patterns.append(fault)
        
        # 规则提取指标数值
        for pattern, metric_type in self.METRIC_PATTERNS.items():
            for match in re.finditer(pattern, text):
                entities.metrics.append({
                    'value': float(match.group(1)),
                    'type': metric_type,
                    'raw': match.group(0)
                })
        
        # Layer 2: 对于规则无法覆盖的实体，使用预训练 NER 模型
        if not entities.components and not entities.fault_patterns:
            ner_results = self.ner_model.predict(text)
            entities.merge(ner_results)
        
        return entities
```

```python
class AgentABTestFramework:
    """
    Agent A/B 测试框架
    
    支持维度：Prompt 版本、模型版本、检索策略
    """
    
    def __init__(self, redis_client):
        self.redis = redis_client
        self.experiments = {}  # active experiments
    
    def assign_variant(self, user_id: str, experiment_id: str) -> str:
        """确定性分流：相同 user_id 始终分到同一组"""
        cache_key = f"ab:{experiment_id}:{user_id}"
        cached = self.redis.get(cache_key)
        if cached:
            return cached
        
        experiment = self.experiments[experiment_id]
        # 基于 user_id hash 的确定性分流
        hash_val = int(hashlib.md5(
            f"{user_id}:{experiment_id}".encode()
        ).hexdigest(), 16) % 100
        
        cumulative = 0
        for variant_name, variant_config in experiment.variants.items():
            cumulative += variant_config.traffic_percent
            if hash_val < cumulative:
                self.redis.setex(cache_key, 86400 * 30, variant_name)
                return variant_name
        
        return "control"
    
    async def record_outcome(self, experiment_id: str, variant: str, 
                            metrics: dict):
        """记录实验结果指标"""
        key = f"ab_results:{experiment_id}:{variant}"
        await self.redis.lpush(key, json.dumps({
            "timestamp": datetime.now().isoformat(),
            "confidence": metrics.get("confidence"),
            "latency_ms": metrics.get("latency_ms"),
            "user_satisfied": metrics.get("user_satisfied"),
            "tokens_used": metrics.get("tokens_used"),
        }))
    
    async def analyze_experiment(self, experiment_id: str) -> ABTestReport:
        """统计分析实验结果（卡方检验 + t 检验）"""
        experiment = self.experiments[experiment_id]
        results = {}
        
        for variant_name in experiment.variants:
            key = f"ab_results:{experiment_id}:{variant_name}"
            raw = await self.redis.lrange(key, 0, -1)
            data = [json.loads(r) for r in raw]
            results[variant_name] = {
                "sample_size": len(data),
                "avg_confidence": np.mean([d["confidence"] for d in data]),
                "avg_latency": np.mean([d["latency_ms"] for d in data]),
                "satisfaction_rate": np.mean([d["user_satisfied"] for d in data]),
                "avg_tokens": np.mean([d["tokens_used"] for d in data]),
            }
        
        # t 检验：treatment vs control
        control = results.get("control", {})
        for variant_name, stats in results.items():
            if variant_name == "control":
                continue
            t_stat, p_value = scipy.stats.ttest_ind(
                [d["confidence"] for d in control_data],
                [d["confidence"] for d in variant_data]
            )
            stats["p_value"] = p_value
            stats["significant"] = p_value < 0.05
        
        return ABTestReport(experiment_id=experiment_id, results=results)
```

### 难点 1：工具返回结果的结构化与标准化 【P0】

**问题本质**

当前 42+ MCP 工具返回**自由格式的 Markdown 文本**。LLM 需要从文本中提取关键信息，本身引入不确定性。

**当前差距**

- ❌ 工具返回格式不统一（有的 Markdown 表格，有的纯文本，有的 JSON 字符串）
- ❌ 异常标注不统一（🔴 / ⚠️ / 文字"异常"混用）
- ❌ 数值精度不统一
- ❌ 缺少机器可读的结构化摘要
- ❌ 工具结果与阈值对比缺失（LLM 需自行判断"92%"是否异常）

**详细优化方案**

```go
// 1. 定义统一的工具返回结构（MCP 协议扩展）
type StructuredToolResult struct {
    // 原始文本（供 LLM 深度分析）
    Text string `json:"text"`
    
    // 结构化摘要（供程序直接消费）
    Summary *ResultSummary `json:"structured_summary,omitempty"`
}

type ResultSummary struct {
    // 整体状态：healthy | warning | critical | unknown
    Status string `json:"status"`
    
    // 关键指标（机器可读）
    KeyMetrics []MetricResult `json:"key_metrics"`
    
    // 检测到的异常列表
    Anomalies []string `json:"anomalies"`
    
    // 一句话建议
    Recommendation string `json:"recommendation,omitempty"`
}

type MetricResult struct {
    Name      string  `json:"name"`        // 指标名
    Value     float64 `json:"value"`       // 当前值
    Unit      string  `json:"unit"`        // 单位：percent, ms, count, bytes
    Threshold float64 `json:"threshold"`   // 阈值
    Status    string  `json:"status"`      // normal | warning | critical
}

// 2. 重构工具实现示例
func (t *HDFSNameNodeStatus) Execute(ctx context.Context, params map[string]interface{}) (*StructuredToolResult, error) {
    status, err := t.client.GetNameNodeStatus(ctx, targetNN)
    if err != nil {
        return nil, err
    }
    
    // 构建结构化摘要
    heapPercent := float64(status.HeapUsed) / float64(status.HeapMax) * 100
    summary := &ResultSummary{
        Status: classifyStatus(heapPercent, 85, 90),
        KeyMetrics: []MetricResult{
            {Name: "heap_percent", Value: heapPercent, Unit: "percent",
             Threshold: 85.0, Status: classifyStatus(heapPercent, 85, 90)},
            {Name: "rpc_queue_length", Value: float64(status.RPCQueueLen), Unit: "count",
             Threshold: 100.0, Status: classifyStatus(float64(status.RPCQueueLen), 100, 200)},
            {Name: "under_replicated_blocks", Value: float64(status.UnderReplicatedBlocks), Unit: "count",
             Threshold: 0, Status: func() string {
                if status.UnderReplicatedBlocks > 0 { return "warning" }
                return "normal"
             }()},
        },
    }
    
    // 自动检测异常
    if heapPercent > 85 {
        summary.Anomalies = append(summary.Anomalies,
            fmt.Sprintf("NN 堆内存 %.1f%% 超过 85%% 警戒线", heapPercent))
    }
    if status.SafeMode {
        summary.Anomalies = append(summary.Anomalies,
            fmt.Sprintf("NameNode 处于 SafeMode: %s", status.SafeModeReason))
        summary.Status = "critical"
    }
    
    if len(summary.Anomalies) > 0 {
        summary.Recommendation = "建议检查 NN 堆内存配置和小文件数量"
    }
    
    return &StructuredToolResult{
        Text:    buildMarkdownText(status),  // 保留人类可读文本
        Summary: summary,
    }, nil
}

func classifyStatus(value, warnThreshold, critThreshold float64) string {
    if value >= critThreshold { return "critical" }
    if value >= warnThreshold { return "warning" }
    return "normal"
}
```

```python
class ConfidenceCalibrator:
    """
    基于历史预测-实际对比的置信度校准
    
    解决问题：Agent 报告 confidence=0.85，但实际只有 65% 正确
    """
    
    def __init__(self):
        self.calibration_model = None  # Isotonic Regression 校准模型
        self.history = []  # (predicted, actual) pairs
    
    def calibrate(self, raw_confidence: float) -> float:
        """将 LLM 输出的原始置信度校准为更准确的概率"""
        if self.calibration_model is None:
            return raw_confidence  # 数据不足时返回原始值
        return float(self.calibration_model.predict([[raw_confidence]])[0])
    
    async def collect_feedback(self, request_id: str, predicted_conf: float, 
                               actual_correct: bool):
        """收集反馈数据用于校准"""
        self.history.append((predicted_conf, 1.0 if actual_correct else 0.0))
        
        # 每积累 100 个样本重新训练校准模型
        if len(self.history) >= 100 and len(self.history) % 50 == 0:
            await self._retrain()
    
    async def _retrain(self):
        """重新训练校准模型"""
        from sklearn.isotonic import IsotonicRegression
        predicted = [h[0] for h in self.history]
        actual = [h[1] for h in self.history]
        
        self.calibration_model = IsotonicRegression(
            y_min=0, y_max=1, out_of_bounds="clip"
        )
        self.calibration_model.fit(predicted, actual)
        
        # 输出校准报告
        logger.info(f"Confidence calibrator retrained with {len(self.history)} samples")
        # 如果发现系统性过度自信 → 告警
        avg_predicted = np.mean(predicted)
        avg_actual = np.mean(actual)
        if avg_predicted - avg_actual > 0.15:
            logger.warning(
                f"系统性过度自信检测: 平均预测 {avg_predicted:.2f} "
                f"vs 实际 {avg_actual:.2f}，差距 {avg_predicted-avg_actual:.2f}"
            )
```

```python
class PriorityTaskScheduler:
    """基于优先级的异步任务调度"""
    
    def __init__(self):
        # 三级优先级队列
        self.queues = {
            "high": asyncio.Queue(),     # 告警处理
            "medium": asyncio.Queue(),   # 用户查询
            "low": asyncio.Queue(),      # 巡检任务
        }
        # 并发控制
        self.semaphores = {
            "high": asyncio.Semaphore(10),    # 告警：最多 10 并发
            "medium": asyncio.Semaphore(5),   # 查询：最多 5 并发
            "low": asyncio.Semaphore(2),      # 巡检：最多 2 并发
        }
    
    async def submit(self, task, priority: str = "medium"):
        await self.queues[priority].put(task)
    
    async def worker(self, priority: str):
        """每个优先级的消费者"""
        while True:
            task = await self.queues[priority].get()
            async with self.semaphores[priority]:
                try:
                    await task.execute()
                except Exception as e:
                    logger.error(f"Task failed [{priority}]: {e}")
                    if priority == "high":
                        # 高优先级任务失败 → 重试一次
                        await self.queues[priority].put(task)

class RetryableJob:
    """带退避重试的定时任务"""
    
    MAX_RETRIES = 3
    
    async def execute_with_retry(self, job_func):
        for attempt in range(self.MAX_RETRIES):
            try:
                await job_func()
                # 成功 → 重置连续失败计数
                self.consecutive_failures = 0
                return
            except Exception as e:
                self.consecutive_failures += 1
                wait_time = min(30 * (2 ** attempt), 300)  # 指数退避，最长 5min
                logger.warning(
                    f"Job failed (attempt {attempt+1}/{self.MAX_RETRIES}), "
                    f"retrying in {wait_time}s: {e}"
                )
                await asyncio.sleep(wait_time)
        
        # 连续 3 次失败
        logger.error(f"Job failed after {self.MAX_RETRIES} retries")
        metrics.job_consecutive_failures.inc()
        
        if self.consecutive_failures >= 3:
            # 连续 3 个周期都失败 → 暂停任务 + 告警
            scheduler.pause_job(self.job_id)
```

```python
class DynamicFewShotSelector:
    """基于语义相似度动态选择最相关的 Few-shot 示例"""
    
    def __init__(self, examples_path: str):
        self.examples = self._load_examples(examples_path)
        self.example_embeddings = self._embed_examples()
    
    def select(self, query: str, max_examples: int = 3, 
               max_tokens: int = 1500) -> list[dict]:
        """选择与当前查询最相关的示例"""
        query_embedding = self.embedding_model.encode(query)
        
        # 计算相似度
        similarities = cosine_similarity(
            [query_embedding], self.example_embeddings
        )[0]
        
        # 按相似度排序
        ranked = sorted(
            enumerate(similarities), key=lambda x: x[1], reverse=True
        )
        
        # 在 Token 预算内选择
        selected = []
        total_tokens = 0
        for idx, score in ranked[:max_examples * 2]:
            example = self.examples[idx]
            example_tokens = self._count_tokens(str(example))
            if total_tokens + example_tokens <= max_tokens:
                selected.append(example)
                total_tokens += example_tokens
            if len(selected) >= max_examples:
                break
        
        return selected

class CostAnomalyDetector:
    """基于 EWMA 的 Token 成本异常检测"""
    
    def __init__(self, span: int = 24):  # 24 小时窗口
        self.ewma_mean = None
        self.ewma_std = None
        self.alpha = 2.0 / (span + 1)
    
    async def check(self, hourly_cost: float) -> bool:
        """检测当前小时成本是否异常"""
        if self.ewma_mean is None:
            self.ewma_mean = hourly_cost
            self.ewma_std = 0
            return False
        
        # 更新 EWMA
        self.ewma_mean = self.alpha * hourly_cost + (1 - self.alpha) * self.ewma_mean
        deviation = abs(hourly_cost - self.ewma_mean)
        self.ewma_std = self.alpha * deviation + (1 - self.alpha) * self.ewma_std
        
        # 3-sigma 检测
        if deviation > 3 * max(self.ewma_std, 0.01):
            await alertmanager.fire(
                "TokenCostAnomaly",
                f"当前小时 Token 成本 ${hourly_cost:.2f} "
                f"偏离均值 ${self.ewma_mean:.2f} 超过 3σ"
            )
            return True
        return False
```
